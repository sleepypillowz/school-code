CREATE TABLE TBL_EMPLOYEES (
    EMPLOYEE_NO BIGINT PRIMARY KEY,
    BIRTH_DATE DATE,
    FIRST_NAME VARCHAR(50),
    LAST_NAME VARCHAR(50),
	  GENDER VARCHAR,
    HIRE_DATE DATE
);

CREATE TABLE TBL_EMP_TITLES (
    EMPLOYEE_NO BIGINT FOREIGN KEY REFERENCES TBL_EMPLOYEES(EMPLOYEE_NO),
    JOB_TITLE VARCHAR(50),
    FROM_DATE DATE,
	  TO_DATE DATE
);

CREATE TABLE TBL_EMP_SALARY (
    EMPLOYEE_NO BIGINT FOREIGN KEY REFERENCES TBL_EMPLOYEES(EMPLOYEE_NO),
    SALARY INT,
    FROM_DATE DATE,
	  TO_DATE DATE
);

CREATE TABLE TBL_DEPT_EMPLOYEES (
    EMPLOYEE_NO BIGINT FOREIGN KEY REFERENCES TBL_EMPLOYEES(EMPLOYEE_NO),
    DEPARTMENT_NO FOREIGN KEY REFERENCES TBL_DEPARTMENTS(DEPARTMENT_NO),
    FROM_DATE DATE,
    TO_DATE DATE
);

CREATE TABLE TBL_DEPARTMENTS (
    DEPARTMENT_NO CHAR(10) PRIMARY KEY,
    DEPARTMENT_NAME VARCHAR(40)
);

CREATE TABLE TBL_DEPT_MANAGERS (
    DEPARTMENT_NO CHAR(10) FOREIGN KEY REFERENCES TBL_DEPARTMENTS(DEPARTMENT_NO),
    FROM_DATE DATE,
    TO_DATE DATE
);
