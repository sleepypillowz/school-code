
CREATE TABLE TBL_GENRE (
	GENRE VARCHAR PRIMARY KEY,
    COMEDY VARCHAR ,
	ROMANCE VARCHAR,
    DRAMA VARCHAR,
	THRILLER VARCHAR,
	SHOUNEN VARCHAR
);

CREATE TABLE TBL_AUTHOR (
	AUTHOR VARCHAR PRIMARY KEY FOREIGN KEY REFERENCES TBL_GENRE(GENRE),
    AKA_AKASAKA VARCHAR,
    NATSUKI_TAKAYA VARCHAR,
    LI_HAOLING VARCHAR,
	YOMI_SARACHI VARCHAR,
	HIROMU_ARAKAWA VARCHAR 
);

CREATE TABLE TBL_FAVANIME (
	FAVANIME VARCHAR FOREIGN KEY REFERENCES TBL_AUTHOR(AUTHOR),
	KAGUYA_SAMA_LOVE_IS_WAR VARCHAR,
	FRUITS_BASKET VARCHAR,
    LINK_CLICK VARCHAR,
	STEINS_GATE VARCHAR,
	FULLMETAL_ALCHEMIST_BROTHERHOOD VARCHAR
);
